function Bullet(args) {
  let {x, y, d, color} = args;
  let assoc = AssocMixin(Bullet, args);
  let merge = MergeMixin(Bullet, args);
  let that = Object.freeze({
    x,
    y,
    w: 5,
    h: 5,
    d,
    color,
    assoc,
    merge,
    update: () => assoc("y", y + d)
  });
  return that;
}

function PlayerBullet(args) {
  let {x, y} = args;
  return Bullet({
    x, y, d: -1, color: 'white'
  });
}

function EnemyBullet(args) {
  let {x, y} = args;
  return Bullet({
    x, y, d: 1, color: '#FF9900'
  });
}
